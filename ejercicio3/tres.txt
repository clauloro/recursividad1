Algoritmo Bandera_Dijkstra

Entrada:
    ca: CADENA
    ficha <- ca 
    ficha_color <- ["Rojo", "Verde", "Azul"]

Precondicion:
    ficha=>1
    ca =! NULO
    longitud (ca) => 1


Funciones:

mientras j <= k hacer:
    si fichas[j] == "Rojo" entonces:
        intercambiar(fichas[i], fichas[j])
        i <- i + 1
        j <- j + 1
        contador_rojas <- contador_rojas + 1
    
    sino si fichas[j] == "Verde" entonces:
        j <- j + 1
        contador_verdes <- contador_verdes + 1
    
    sino si fichas[j] == "Azul" entonces:
        intercambiar(fichas[j], fichas[k])
        k <- k - 1
        contador_azules <- contador_azules + 1
    
    sino:
        # En este caso la ficha no pertenece a ninguno de los colores
        j <- j + 1

sublista_rojas <- lista_vacia
sublista_verdes <- lista_vacia
sublista_azules <- lista_vacia

para indice en rango(0, contador_rojas) hacer:
    sublista_rojas.agregar("Rojo")

para indice en rango(0, contador_verdes) hacer:
    sublista_verdes.agregar("Verde")

para indice en rango(0, contador_azules) hacer:
    sublista_azules.agregar("Azul")

resultado <- sublista_rojas + sublista_verdes + sublista_azules
